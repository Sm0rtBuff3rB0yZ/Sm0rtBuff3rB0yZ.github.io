I love big RSA moduli, because I think that they are sufficiently secure. How about a 10240-bit one ? You'll never be able to crack my message! b0br4nks loves big RSA moduli.

source.py:
import random
from Crypto.Util.number import isPrime, sieve_base as primes
from flag import flag

def generate_large_prime(bits):
    while True:
        candidate = random.getrandbits(bits)
        candidate |= 1
        if isPrime(candidate):
            return candidate

def encrypt_message(message, exponent, modulus):
    m = int.from_bytes(message.encode(), 'big')
    return pow(m, exponent, modulus)

e = 0x10001
p = generate_large_prime(2048)
q = generate_large_prime(2048)
n = p * q
encrypted_flag = encrypt_message(flag, e, n)
